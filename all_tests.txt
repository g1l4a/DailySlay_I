    if a > b then
        return a
    else

    end
end

print(result)

TEST 10
routine factorial(n: integer): integer is
    if n = 0 then
        return 1
    else
        return n * factorial(n - 1)
    end
end
var fact: integer
fact := factorial(5)
print(fact)
